Entry point for executing the JetLeaf CLI process.

  • args: The command-line arguments passed by the user.

Implementations should perform initialization, argument parsing, and dispatch the request to an appropriate CommandRunner.

@override
Future<void> run(List<String> args) async {
  print('üçÉ JetLeaf CLI - Command Line Interface');
  logger.space();

  if (args.isEmpty || (args.length == 1 && args.first.equalsAny(['--help', '-h']))) {
    logger.warn('No command specified. Showing general help.');
    logger.space();
    print(usage);
  } else {
    final commandName = args.first;
    try {
      final targetCommand = RUNNERS.firstWhere((cmd) => cmd.command == commandName);
      print(targetCommand.usage);
    } catch (e) {
      logger.error('‚ùå Unknown command: $commandName');
      logger.space();
      print(usage);
    }
  }
}
